					type = industrial_complex

			factor = 10
		}
	}

	SOV_strengthen_fortifications = {
		icon = generic_fortify_the_borders
		
		available = {
			has_political_power > 75
			has_war_support > 0.3
			OR = {
				has_completed_focus = SOV_border_fortifications
				has_completed_focus = SOV_finnish_security
				has_completed_focus = SOV_eastern_poland
			}
		}
		
		visible = {
			OR = {
				has_completed_focus = SOV_border_fortifications
				has_completed_focus = SOV_finnish_security
				has_completed_focus = SOV_eastern_poland
			}
		}
		
		cost = 75
		
		days_remove = 70
		
		modifier = {
			production_speed_bunker_factor = 0.25
			production_speed_coastal_bunker_factor = 0.25
		}
		
		remove_effect = {
			if = {
				limit = { has_completed_focus = SOV_border_fortifications }
				random_owned_state = {
					limit = {
						is_core_of = SOV
						is_controlled_by = ROOT
						is_on_continent = asia
					}
					add_building_construction = {
						type = bunker
						level = 2
						instant_build = yes
						province = {
							all_provinces = yes
							limit_to_border = yes
						}
					}
				}
			}
			if = {
				limit = { has_completed_focus = SOV_finnish_security }
				random_owned_state = {
					limit = {
						is_core_of = SOV
						is_controlled_by = ROOT
						is_on_continent = europe
						any_neighbor_state = {
							is_owned_by = FIN
						}
					}
					add_building_construction = {
						type = bunker
						level = 2
						instant_build = yes
						province = {
							all_provinces = yes
							limit_to_border = yes
						}
					}
				}
			}
			if = {
				limit = { has_completed_focus = SOV_eastern_poland }
				random_owned_state = {
					limit = {
						is_core_of = SOV
						is_controlled_by = ROOT
						is_on_continent = europe
						any_neighbor_state = {
							is_owned_by = POL
						}
					}
					add_building_construction = {
						type = bunker
						level = 2
						instant_build = yes
						province = {
							all_provinces = yes
							limit_to_border = yes
						}
					}
				}
			}
		}
		
		ai_will_do = {
			factor = 5
			modifier = {
				factor = 3
				any_neighbor_country = {
					has_war_with = ROOT
				}
			}
		}
	}
}
